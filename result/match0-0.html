import java.util.*;
public class Matjaz {

    public static void main(String[] args) {
        
        int n;
        int m;
        
        int[] a;
        int[] b;
        
        int dolzina;
        int[] tabela;
        
        Scanner sc = new Scanner(System.in);
        
        n = sc.nextInt();
        a = new int[n];
        m = sc.nextInt();
        b = new int[m];
        
        dolzina = n + m;
        tabela = new int [dolzina];
        
        for(int i = 0; i < n; i++){
            a[i] = sc.nextInt();
        }
        
        for(int i = 0; i < m; i++){
            b[i] = sc.nextInt();
        }
        
        Izziv3 zlij = new Izziv3();
        tabela = zlij.zlivanje(a,b,tabela);
        
        System.out.println("");
        for(int i = 0; i < tabela.length; i++){
            System.out.println(tabela[i] + "");
        }
        
    }
    
    public int[] zlivanje(int[] a, int[] b, int[] tabela){
        int i = 0;
        int j = 0;
        int index = 0;
        
        while(i!= a.length|| j != b.length){
            if(i > a.length-1){
                System.out.print("b");
                tabela[index] = b[j];
                j++;
            }
            else 
                if(j > b.length-1){
                System.out.print("a");
                tabela[index] = a[i];
            }
            else{
                if(a[i] <= b[j]){
                    System.out.print("a");
                    tabela[index] = a[i];
                    i++;
                }
                else{
                    System.out.print("b");
                    tabela[index] = b[j];
                    j++;
                }
                index++;    
            }
        }
        return tabela;
    }
    
}
